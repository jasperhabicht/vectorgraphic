\clist_new:N \l_qdraw_arrow_start_keys_clist
\clist_new:N \l_qdraw_arrow_end_keys_clist

\keys_define:nn { qdraw / path } {
    arrow ~ start           .code:n         = {
        \tl_put_right:Nn \l_qdraw_path_postactions_tl {
            \fp_if_nan:nF { \l_qdraw_path_first_slope_fp } {
                \qdraw_put_arrow:nVVV {#1}
                    \l_qdraw_path_first_point_fp
                    \l_qdraw_path_first_slope_fp
                    \l_qdraw_arrow_start_keys_clist
            }
        }
    } ,
    arrow ~ start           .default:n      = { default } ,
    arrow ~ end             .code:n         = {
        \tl_put_right:Nn \l_qdraw_path_postactions_tl {
            \fp_if_nan:nF { \l_qdraw_path_last_slope_fp } {
                \qdraw_put_arrow:nVVV {#1}
                    \l_qdraw_path_last_point_fp
                    \l_qdraw_path_last_slope_fp
                    \l_qdraw_arrow_end_keys_clist
            }
        }
    } ,
    arrow ~ end             .default:n      = { default } ,
    arrow ~ start ~ style   .clist_set:N    = \l_qdraw_arrow_start_keys_clist ,
    arrow ~ end ~ style     .clist_set:N    = \l_qdraw_arrow_end_keys_clist ,
    arrow ~ style           .code:n         = {
        \clist_set:Nn \l_qdraw_arrow_start_keys_clist {#1}
        \clist_set:Nn \l_qdraw_arrow_end_keys_clist {#1}
    } ,
    arrow ~ style           .initial:n      = { fill }
}

\cs_new_protected:Npn \qdraw_put_arrow:nnnn #1#2#3#4 {
    \draw_scope_begin:
        \draw_transform_shift:n {#2}
        \draw_transform_rotate:n {#3}
        \clist_clear:N \l_qdraw_path_use_clist
        \keys_set:nn { qdraw / path } {#4}
        \qdraw_set_path_options:
        \use:c { qdraw_arrow_ #1 : }
        \draw_path_use_clear:e { \clist_use:N \l_qdraw_path_use_clist }
    \draw_scope_end:
}
\cs_generate_variant:Nn \qdraw_put_arrow:nnnn { nVVV }

\cs_new_protected:Npn \qdraw_arrow_none: { }

\cs_new_protected:Npn \qdraw_arrow_default: {
    \draw_path_moveto:n { -2.5 , -5 }
    \draw_path_curveto:nn { -1 , -3 } { 0 , 0 }
    \draw_path_curveto:nn { 1 , -3 } { 2.5 , -5 }
    \draw_path_close:
}

\cs_new_protected:Npn \qdraw_arrow_stealth: {
    \draw_path_moveto:n { 0 , 0 }
    \draw_path_lineto:n { -2.5 , -5 }
    \draw_path_lineto:n { 0 , -3.5 }
    \draw_path_lineto:n { 2.5 , -5 }
    \draw_path_close:
}

\cs_new_protected:Npn \qdraw_arrow_triangle: {
    \draw_path_moveto:n { 0 , 0 }
    \draw_path_lineto:n { -2.5 , -5 }
    \draw_path_lineto:n { 2.5 , -5 }
    \draw_path_close:
}

\cs_new_protected:Npn \qdraw_arrow_circle: {
    \draw_path_circle:nn { 0 , -2.5 } { 2.5 }
}

\cs_new_protected:Npn \qdraw_arrow_simple: {
    \draw_path_moveto:n { -3 , -3 }
    \draw_path_curveto:nn { -0.75 , -0.75 } { 0 , 0 }
    \draw_path_curveto:nn { 0.75 , -0.75 } { 3 , -3 }
}

\cs_new_protected:Npn \qdraw_arrow_straight: {
    \draw_path_moveto:n { -3 , -3 }
    \draw_path_lineto:n { 0 , 0 }
    \draw_path_lineto:n { 3 , -3 }
}

\cs_new_protected:Npn \qdraw_arrow_bar: {
    \draw_path_moveto:n { -3 , 0 }
    \draw_path_lineto:n { 3 , 0 }
}
