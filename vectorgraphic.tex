\documentclass{article}
\usepackage{vectorgraphic}

% syntax cheatsheet:
% [<options>] > set options for the path (can be used after every coordinate)
% (<a>,<b>) > coordinate with x and y parts
% (<a>:<b>) > coordinate with angle and radius parts
% -- (<a>,<b>) > draw a line to the next coordinate
% .. (<a>,<b>) .. (<c>,<d>) > draw a curve to the next coordinate with single control point
% .. (<a>,<b>) & (<c>,<d>) .. (<e>,<f>) > draw a curve to the next coordinate with two control points
% ! > close the path

\begin{document}

\begin{vectorgraphic}
    \path[stroke, stroke color=blue] (0,0) -- (0cm,1cm) .. (1cm,1cm) .. (1cm,2cm) -- (1.5cm,2.5cm) -- (1cm,0cm) ! ;
\end{vectorgraphic}

\begin{vectorgraphic}
    \path[fill, fill color=red] (0,0) -- (0cm,1cm) .. (1cm,1cm) & (1cm,2cm) .. (2cm,2cm) -- (3cm,2.5cm) -- (2cm,0cm) ! ;
\end{vectorgraphic}

\end{document}
