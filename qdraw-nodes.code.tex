\keys_define:nn { qdraw / path / node } {
    at                      .fp_set:N       = \l_qdraw_node_at_fp ,
    name                    .tl_set:N       = \l_qdraw_node_name_tl ,
    anchor                  .clist_set:N    = \l_qdraw_node_anchor_clist ,
    anchor                  .initial:n      = { hc , vc } ,
    width                   .code:n         = {
        \bool_set_true:N \l__qdraw_node_vbox_bool
        \dim_set:Nn \l_qdraw_node_width_dim {#1}
    } ,
    padding                 .clist_set:N    = \l_qdraw_node_padding_clist ,
    padding                 .initial:n      = { 5pt } ,
    text ~ color            .str_set:N      = \l_qdraw_node_text_color_str ,
    text ~ color            .initial:n      = { . } ,
    frame                   .str_set:N      = \l_qdraw_node_frame_str ,
    frame                   .initial:n      = { rectangle } ,
}

\cs_new_protected:cpn { __qdraw_process_path_ n :w } node #1#2 \s__qdraw_stop {
    \cs_if_exist_use:cTF {
        __qdraw_process_path_node_ \tl_head:n {#1} :w
    } {
        #1#2 \s__qdraw_stop
    } {
        \qdraw_put_node:nn { } {#1}
        \tl_trim_spaces_apply:nN {#2} \__qdraw_continue_path:n
    }
}

\cs_new_protected:cpn { __qdraw_process_path_node_ [ :w } [ #1 ] #2#3 \s__qdraw_stop {
    \qdraw_put_node:nn {#1} {#2}
    \tl_trim_spaces_apply:nN {#3} \__qdraw_continue_path:n
}

\cs_new_protected:Npn \qdraw_put_node:nn #1#2 {
    \draw_scope_begin:
        \fp_set_eq:NN \l_qdraw_node_at_fp \g_qdraw_path_last_point_fp
        \clist_clear:N \l__qdraw_keys_unprocessed_clist
        \keys_set_known:nnN { qdraw / path / node } {#1}
            \l__qdraw_keys_unprocessed_clist
        \tl_if_blank:nF \l_qdraw_node_name_tl {
            \prop_gput:cVV { g__qdraw_points_ \int_use:N \g_qdraw_id_int _prop }
                \l_qdraw_node_name_tl
                \l_qdraw_node_at_fp
        }
        \bool_if:NTF \l__qdraw_node_vbox_bool {
            \vcoffin_set:Nnn \l_qdraw_node_coffin {
                \l_qdraw_node_width_dim
            } {
                \color_select:V \l_qdraw_node_text_color_str
                \normalfont #2
            }
        } {
            \hcoffin_set:Nn \l_qdraw_node_coffin {
                \color_select:V \l_qdraw_node_text_color_str
                \normalfont #2
            }
        }
        \__qdraw_set_node_padding:
        \__qdraw_set_node_frame:V \l__qdraw_keys_unprocessed_clist
        \coffin_attach:NnnNnnnn
            \l_qdraw_node_frame_coffin { hc } { vc }
            \l_qdraw_node_coffin { hc } { vc }
            { 0.5 \l__qdraw_node_padding_left_dim - 0.5 \l__qdraw_node_padding_right_dim }
            { 0.5 \l__qdraw_node_padding_bottom_dim - 0.5 \l__qdraw_node_padding_top_dim }
        \exp_args:Ne
        \draw_coffin_use:NeeV \l_qdraw_node_frame_coffin {
            \clist_item:Nn \l_qdraw_node_anchor_clist { 1 }
        } {
            \clist_item:Nn \l_qdraw_node_anchor_clist { 2 }
        } \l_qdraw_node_at_fp
    \draw_scope_end:
}

\cs_new_protected:Npn \__qdraw_set_node_frame:n #1 {
    \hcoffin_set:Nn \l_qdraw_node_frame_coffin {
        \draw_suspend_begin:
            \draw_begin:
                \keys_set:nn { qdraw / path } {#1}
                \qdraw_set_path_options:
                \use:c { qdraw_node_frame_ \l_qdraw_node_frame_str :NVVVV }
                    \l_qdraw_node_coffin
                    \l__qdraw_node_padding_left_dim
                    \l__qdraw_node_padding_right_dim
                    \l__qdraw_node_padding_top_dim
                    \l__qdraw_node_padding_bottom_dim
                \clist_remove_duplicates:N \l_qdraw_path_use_clist
                \draw_path_use_clear:e { \clist_use:N \l_qdraw_path_use_clist }
            \draw_end:
        \draw_suspend_end:
    }
}
\cs_generate_variant:Nn \__qdraw_set_node_frame:n { V }

\cs_new_protected:Npn \qdraw_node_create_frame:nn #1#2 {
    \cs_new_protected:cpn { qdraw_node_frame_ #1 :Nnnnn } ##1##2##3##4##5 {
        \dim_set:Nn \qdraw_node_natural_width_dim {
            \coffin_wd:N ##1 + ##2 + ##3
        }
        \dim_set:Nn \qdraw_node_natural_height_dim {
            \coffin_ht_plus_dp:N ##1 + ##4 + ##5
        }    
        #2
    }
    \cs_generate_variant:cn { qdraw_node_frame_ #1 :Nnnnn } { NVVVV }
}

\qdraw_node_create_frame:nn { rectangle } {
    \draw_path_rectangle:nn {
        (
            -0.5 * \qdraw_node_natural_width_dim
        ,
            -0.5 * \qdraw_node_natural_width_dim
        )
    } {
        (
            \qdraw_node_natural_width_dim
        ,
            \qdraw_node_natural_height_dim
        )
    }
}

\qdraw_node_create_frame:nn { ellipse } {
    \draw_path_ellipse:nnn { 0pt , 0pt } {
        ( 0.5 * \qdraw_node_natural_width_dim , 0pt )
    } {
        ( 0pt , 0.5 * \qdraw_node_natural_height_dim )
    }
}

\qdraw_node_create_frame:nn { circle } {
    \draw_path_circle:nn { 0pt , 0pt } {
        max(
            0.5 * \qdraw_node_natural_width_dim
        ,
            0.5 * \qdraw_node_natural_height_dim
        )
    }
}

\qdraw_node_create_frame:nn { diamond } {
    \draw_path_moveto:n {
        ( 0pt , \qdraw_node_natural_height_dim )
    }
    \draw_path_lineto:n {
        ( \qdraw_node_natural_width_dim , 0pt )
    }
    \draw_path_lineto:n {
        ( 0pt , -1 * \qdraw_node_natural_height_dim )
    }
    \draw_path_lineto:n {
        ( -1 * \qdraw_node_natural_width_dim , 0pt )
    }
    \draw_path_close:
}
